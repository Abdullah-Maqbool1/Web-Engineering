Server.c:
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <pthread.h>
#include <arpa/inet.h>

#define PORT 8080

void *handleClient(void *arg)
{
    int clientSocket = *(int *)arg;
    free(arg);

    char alert[100];
    char response[200];

    recv(clientSocket, alert, sizeof(alert), 0);

    printf("\nAlert Received: %s\n", alert);

    if (strcmp(alert, "FIRE") == 0)
    {
        strcpy(response,
               "Action: Fire Alarm Activated & Fire Brigade Notified");
    }
    else if (strcmp(alert, "INTRUSION") == 0)
    {
        strcpy(response,
               "Action: Doors Locked & Security Team Notified");
    }
    else if (strcmp(alert, "GASLEAK") == 0)
    {
        strcpy(response,
               "Action: Gas Valve Closed & Ventilation Started");
    }
    else if (strcmp(alert, "PANIC") == 0)
    {
        strcpy(response,
               "Action: Emergency Response Team Dispatched");
    }
    else
    {
        strcpy(response,
               "Unknown Alert Type");
    }

    send(clientSocket, response, strlen(response) + 1, 0);

    close(clientSocket);
    pthread_exit(NULL);
}

int main()
{
    int serverSocket;
    struct sockaddr_in serverAddr;

    serverSocket = socket(AF_INET, SOCK_STREAM, 0);

    serverAddr.sin_family = AF_INET;
    serverAddr.sin_port = htons(PORT);
    serverAddr.sin_addr.s_addr = INADDR_ANY;

    bind(serverSocket,
         (struct sockaddr *)&serverAddr,
         sizeof(serverAddr));

    listen(serverSocket, 5);

    printf("Security Server Started...\n");
    printf("Listening on Port %d\n", PORT);

    while (1)
    {
        int clientSocket;
        struct sockaddr_in clientAddr;
        socklen_t clientLen = sizeof(clientAddr);

        clientSocket = accept(serverSocket,
                              (struct sockaddr *)&clientAddr,
                              &clientLen);

        printf("\nClient Connected\n");

        pthread_t tid;

        int *newSocket = malloc(sizeof(int));
        *newSocket = clientSocket;

        pthread_create(&tid,
                       NULL,
                       handleClient,
                       (void *)newSocket);

        pthread_detach(tid);
    }

    close(serverSocket);

    return 0;
}
